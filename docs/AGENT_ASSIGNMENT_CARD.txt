================================================================================
                         COMMON TEST PATTERNS
================================================================================

PATTERN 1: Basic Unit Test
```python
def test_method_does_something(self):
    """Test that method does X."""
    obj = MyClass()
    result = obj.method()
    assert result == expected_value
```

PATTERN 2: Error Handling
```python
def test_method_raises_on_error(self):
    """Test that method raises on error."""
    with pytest.raises(ValueError):
        obj.method(invalid=True)
```

PATTERN 3: Parameterized
```python
@pytest.mark.parametrize("input,expected", [(1, 2), (2, 4)])
def test_multiple_inputs(self, input, expected):
    assert obj.double(input) == expected
```

PATTERN 4: Mocking
```python
@patch('module.external_func')
def test_with_mock(self, mock_func):
    mock_func.return_value = "mocked"
    result = obj.method()
    assert result == "expected"
```

PATTERN 5: Async
```python
@pytest.mark.asyncio
async def test_async_method(self):
    result = await obj.async_method()
    assert result is not None
```
